baudrate=115200
bootargs=console=ttyS12,115200n8 root=/dev/ram rw earlycon
boota=setenv bootpart 2; setenv rootfs rofs-a; run setmmcargs; ext4load scsi 0:${bootpart} ${loadaddr} fitImage && bootm; echo Error loading kernel FIT image
bootb=setenv bootpart 3; setenv rootfs rofs-b; run setmmcargs; ext4load scsi 0:${bootpart} ${loadaddr} fitImage && bootm; echo Error loading kernel FIT image
bootcmd=run bootmmc
bootdelay=2
bootmmc=if test "${bootside}" = "b"; then run bootb; run boota; else run boota; run bootb; fi
bootside=a
bootspi=fdt addr 100220000 && fdt header get fitsize totalsize && cp.b 100220000 ${loadaddr} ${fitsize} && bootm; echo Error loading kernel FIT image
loadaddr=0x403000000
rootfs=rofs-a
setmmcargs=setenv bootargs ${bootargs} rootwait root=PARTLABEL=${rootfs}
verify=yes
